readLongLe

fun readLongLe(): Long

Overrides BufferedSource.readLongLe

Removes eight bytes from this source and returns a little-endian long.

Buffer buffer = new Buffer()
    .writeByte(0xff)
    .writeByte(0xff)
    .writeByte(0xff)
    .writeByte(0xff)
    .writeByte(0xff)
    .writeByte(0xff)
    .writeByte(0xff)
    .writeByte(0x7f)
    .writeByte(0x0f)
    .writeByte(0x00)
    .writeByte(0x00)
    .writeByte(0x00)
    .writeByte(0x00)
    .writeByte(0x00)
    .writeByte(0x00)
    .writeByte(0x00);
assertEquals(16, buffer.size());

assertEquals(9223372036854775807L, buffer.readLongLe());
assertEquals(8, buffer.size());

assertEquals(15, buffer.readLongLe());
assertEquals(0, buffer.size());
